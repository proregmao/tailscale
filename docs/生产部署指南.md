## 🏗️ 部署方式

### 方式一: 二进制部署 (推荐)

#### 1. 下载预编译二进制文件
```bash
# 下载后端服务
wget https://github.com/your-repo/tailscale-unlimited/releases/latest/unlimited-control
chmod +x unlimited-control

# 下载前端文件
wget https://github.com/your-repo/tailscale-unlimited/releases/latest/headscale-ui.tar.gz
tar -xzf headscale-ui.tar.gz
```

#### 2. 创建配置文件
```bash
# 创建配置目录
mkdir -p /etc/tailscale-unlimited
mkdir -p /var/lib/tailscale-unlimited
mkdir -p /var/log/tailscale-unlimited

# 创建配置文件
cat > /etc/tailscale-unlimited/config.yaml << EOF
# Tailscale Unlimited Control 配置文件
server:
  listen: "0.0.0.0:8081"
  debug: false
  
database:
  path: "/var/lib/tailscale-unlimited/database.db"
  
logging:
  level: "info"
  file: "/var/log/tailscale-unlimited/server.log"
  
security:
  cors_origins: ["http://localhost:5173", "https://your-domain.com"]
  
alerts:
  enabled: true
  check_interval: "30s"
EOF
```

#### 3. 创建systemd服务
```bash
cat > /etc/systemd/system/tailscale-unlimited.service << EOF
[Unit]
Description=Tailscale Unlimited Control Server
After=network.target

[Service]
Type=simple
User=tailscale
Group=tailscale
WorkingDirectory=/opt/tailscale-unlimited
ExecStart=/opt/tailscale-unlimited/unlimited-control -listen :8081
Restart=always
RestartSec=5
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target
EOF
```

#### 4. 启动服务
```bash
# 创建用户
useradd -r -s /bin/false tailscale

# 设置权限
chown -R tailscale:tailscale /var/lib/tailscale-unlimited
chown -R tailscale:tailscale /var/log/tailscale-unlimited

# 启动服务
systemctl daemon-reload
systemctl enable tailscale-unlimited
systemctl start tailscale-unlimited
```

### 方式二: Docker部署

#### 1. 创建Docker Compose文件
```yaml
version: '3.8'

services:
  tailscale-unlimited:
    image: tailscale-unlimited:latest
    container_name: tailscale-unlimited
    ports:
      - "8081:8081"
    volumes:
      - ./data:/data
      - ./logs:/logs
    environment:
      - LISTEN_ADDR=0.0.0.0:8081
      - DB_PATH=/data/database.db
      - LOG_LEVEL=info
    restart: unless-stopped
    
  nginx:
    image: nginx:alpine
    container_name: tailscale-unlimited-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/nginx.conf
      - ./ssl:/etc/nginx/ssl
      - ./headscale-ui:/usr/share/nginx/html
    depends_on:
      - tailscale-unlimited
    restart: unless-stopped
```

#### 2. 创建Nginx配置
```nginx
events {
    worker_connections 1024;
}

http {
    upstream backend {
        server tailscale-unlimited:8081;
    }
    
    server {
        listen 80;
        server_name your-domain.com;
        
        # 重定向到HTTPS
        return 301 https://$server_name$request_uri;
    }
    
    server {
        listen 443 ssl http2;
        server_name your-domain.com;
        
        ssl_certificate /etc/nginx/ssl/cert.pem;
        ssl_certificate_key /etc/nginx/ssl/key.pem;
        
        # 前端静态文件
        location / {
            root /usr/share/nginx/html;
            try_files $uri $uri/ /index.html;
        }
        
        # API代理
        location /api/ {
            proxy_pass http://backend;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
        
        # Tailscale机器注册端点
        location /machine/ {
            proxy_pass http://backend;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
    }
}
```

#### 3. 启动Docker服务
```bash
docker-compose up -d
```

## 🔄 备份和恢复

### 1. 数据备份
```bash
#!/bin/bash
# 备份脚本
BACKUP_DIR="/backup/tailscale-unlimited"
DATE=$(date +%Y%m%d_%H%M%S)

mkdir -p $BACKUP_DIR

# 备份数据库
cp /var/lib/tailscale-unlimited/database.db $BACKUP_DIR/database_$DATE.db

# 备份配置
cp /etc/tailscale-unlimited/config.yaml $BACKUP_DIR/config_$DATE.yaml

# 压缩备份
tar -czf $BACKUP_DIR/backup_$DATE.tar.gz $BACKUP_DIR/*_$DATE.*

# 清理旧备份 (保留30天)
find $BACKUP_DIR -name "backup_*.tar.gz" -mtime +30 -delete
```

### 2. 自动备份
```bash
# 添加到crontab
crontab -e

# 每日凌晨2点备份
0 2 * * * /opt/tailscale-unlimited/backup.sh
```

### 3. 数据恢复
```bash
# 停止服务
systemctl stop tailscale-unlimited

# 恢复数据库
cp /backup/tailscale-unlimited/database_YYYYMMDD_HHMMSS.db /var/lib/tailscale-unlimited/database.db

# 恢复配置
cp /backup/tailscale-unlimited/config_YYYYMMDD_HHMMSS.yaml /etc/tailscale-unlimited/config.yaml

# 启动服务
systemctl start tailscale-unlimited
```
