### 第二阶段：日志管理页面Svelte化

#### 2.1 高级功能实现

**实时日志流**：
```svelte
<script lang="ts">
  import { onMount } from 'svelte';
  import { writable } from 'svelte/store';
  
  // 日志流状态
  let autoRefresh = true;
  let refreshInterval = 5000;
  let logs = [];
  let filteredLogs = [];
  
  // 过滤器
  let filters = {
    level: '',
    component: '',
    search: '',
    timeRange: '1h'
  };
  
  // 响应式过滤
  $: filteredLogs = logs.filter(log => {
    if (filters.level && log.level !== filters.level) return false;
    if (filters.component && log.component !== filters.component) return false;
    if (filters.search && !log.message.toLowerCase().includes(filters.search.toLowerCase())) return false;
    return true;
  });
  
  // 虚拟滚动（处理大量日志）
  let virtualList;
  let visibleLogs = [];
  
  $: if (filteredLogs.length > 1000) {
    // 实现虚拟滚动
    visibleLogs = filteredLogs.slice(0, 100);
  } else {
    visibleLogs = filteredLogs;
  }
  
  // WebSocket连接（实时日志）
  let ws;
  onMount(() => {
    if (autoRefresh) {
      connectWebSocket();
    }
    
    return () => {
      if (ws) ws.close();
    };
  });
  
  function connectWebSocket() {
    ws = new WebSocket('ws://localhost:8080/api/v1/logs/stream');
    ws.onmessage = (event) => {
      const newLog = JSON.parse(event.data);
      logs = [newLog, ...logs].slice(0, 10000); // 保持最新10000条
    };
  }
</script>

<div class="container mx-auto px-4 py-8">
  <!-- 实时控制 -->
  <div class="bg-white rounded-lg shadow p-4 mb-6">
    <div class="flex items-center justify-between">
      <div class="flex items-center space-x-4">
        <label class="flex items-center">
          <input type="checkbox" bind:checked={autoRefresh} class="mr-2">
          实时更新
        </label>
        <select bind:value={refreshInterval} class="px-3 py-1 border rounded">
          <option value={1000}>1秒</option>
          <option value={5000}>5秒</option>
          <option value={10000}>10秒</option>
        </select>
      </div>
      <div class="text-sm text-gray-500">
        显示 {visibleLogs.length} / {filteredLogs.length} 条日志
      </div>
    </div>
  </div>
  
  <!-- 高级过滤器 -->
  <div class="bg-white rounded-lg shadow p-4 mb-6">
    <div class="grid grid-cols-1 md:grid-cols-4 gap-4">
      <input 
        type="text" 
        bind:value={filters.search}
        placeholder="搜索日志内容..."
        class="px-3 py-2 border rounded-md"
      >
      <select bind:value={filters.level} class="px-3 py-2 border rounded-md">
        <option value="">所有级别</option>
        <option value="error">错误</option>
        <option value="warn">警告</option>
        <option value="info">信息</option>
        <option value="debug">调试</option>
      </select>
      <select bind:value={filters.component} class="px-3 py-2 border rounded-md">
        <option value="">所有组件</option>
        <option value="api">API</option>
        <option value="auth">认证</option>
        <option value="device">设备</option>
        <option value="alert">告警</option>
      </select>
      <select bind:value={filters.timeRange} class="px-3 py-2 border rounded-md">
        <option value="1h">最近1小时</option>
        <option value="6h">最近6小时</option>
        <option value="24h">最近24小时</option>
        <option value="7d">最近7天</option>
      </select>
    </div>
  </div>
  
  <!-- 日志列表 -->
  <div class="bg-white rounded-lg shadow">
    <div class="overflow-hidden">
      {#each visibleLogs as log (log.id)}
        <div class="border-b border-gray-200 p-4 hover:bg-gray-50" transition:slide>
          <div class="flex items-start space-x-3">
            <span class="inline-flex px-2 py-1 text-xs font-semibold rounded-full {getLevelClass(log.level)}">
              {log.level}
            </span>
            <div class="flex-1 min-w-0">
              <div class="text-sm text-gray-900">{log.message}</div>
              <div class="text-xs text-gray-500 mt-1">
                {formatDate(log.created_at)} • {log.component}
              </div>
            </div>
          </div>
        </div>
      {/each}
    </div>
  </div>
</div>
```

```svelte
<script lang="ts">
  import { onMount, afterUpdate } from 'svelte';
  import Chart from 'chart.js/auto';
  
  // 图表实例
  let usageChart;
  let performanceChart;
  let deviceChart;
  
  // 数据状态
  let activeTab = 'usage';
  let reportData = {
    usage: {},
    performance: {},
    devices: {}
  };
  
  // 图表配置
  const chartConfigs = {
    usage: {
      type: 'line',
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          title: {
            display: true,
            text: '设备在线趋势'
          }
        },
        scales: {
          y: {
            beginAtZero: true
          }
        }
      }
    },
    performance: {
      type: 'doughnut',
      options: {
        responsive: true,
        maintainAspectRatio: false,
        plugins: {
          title: {
            display: true,
            text: '系统资源使用情况'
          }
        }
      }
    }
  };
  
  // 响应式图表更新
  $: if (reportData.usage.timeline && usageChart) {
    usageChart.data.datasets[0].data = reportData.usage.timeline;
    usageChart.update('none'); // 无动画更新
  }
  
  // 图表初始化
  function initCharts() {
    // 使用图表
    const usageCtx = document.getElementById('usageChart');
    if (usageCtx) {
      usageChart = new Chart(usageCtx, {
        ...chartConfigs.usage,
        data: {
          labels: getTimeLabels(),
          datasets: [{
            label: '在线设备数',
            data: reportData.usage.timeline || [],
            borderColor: 'rgb(59, 130, 246)',
            backgroundColor: 'rgba(59, 130, 246, 0.1)',
            tension: 0.1
          }]
        }
      });
    }
    
    // 性能图表
    const perfCtx = document.getElementById('performanceChart');
    if (perfCtx) {
      performanceChart = new Chart(perfCtx, {
        ...chartConfigs.performance,
        data: {
          labels: ['CPU', '内存', '磁盘'],
          datasets: [{
            data: [
              reportData.performance.cpu_usage || 0,
              reportData.performance.memory_usage || 0,
              reportData.performance.disk_usage || 0
            ],
            backgroundColor: [
              'rgba(59, 130, 246, 0.8)',
              'rgba(16, 185, 129, 0.8)',
              'rgba(245, 158, 11, 0.8)'
            ]
          }]
        }
      });
    }
  }
  
  // 数据导出功能
  async function exportReport(type, format) {
    try {
      const response = await fetch(`/api/v1/reports/export/${type}?format=${format}`);
      const blob = await response.blob();
      
      const url = window.URL.createObjectURL(blob);
      const a = document.createElement('a');
      a.href = url;
      a.download = `${type}_report_${new Date().toISOString().split('T')[0]}.${format}`;
      document.body.appendChild(a);
      a.click();
      document.body.removeChild(a);
      window.URL.revokeObjectURL(url);
    } catch (error) {
      console.error('Export failed:', error);
      alert('导出失败，请重试');
    }
  }
  
  onMount(async () => {
    await loadReportData();
    initCharts();
  });
  
  afterUpdate(() => {
    // 确保图表在DOM更新后正确渲染
    if (activeTab === 'usage' && !usageChart) {
      initCharts();
    }
  });
</script>

<div class="container mx-auto px-4 py-8">
  <!-- 标签页导航 -->
  <div class="bg-white rounded-lg shadow mb-6">
    <div class="border-b border-gray-200">
      <nav class="-mb-px flex space-x-8 px-6" aria-label="Tabs">
        <button 
          on:click={() => activeTab = 'usage'}
          class="py-4 px-1 border-b-2 font-medium text-sm {activeTab === 'usage' ? 'border-blue-500 text-blue-600' : 'border-transparent text-gray-500 hover:text-gray-700'}"
        >
          使用情况报表
        </button>
        <button

          class="py-4 px-1 border-b-2 font-medium text-sm {activeTab === 'performance' ? 'border-blue-500 text-blue-600' : 'border-transparent text-gray-500 hover:text-gray-700'}"
        >
```

#### Modal.svelte
```svelte
<script lang="ts">
  import { createEventDispatcher } from 'svelte';
  import { fade, scale } from 'svelte/transition';
  
  export let title = '';
  export let size = 'md'; // sm, md, lg, xl
  
  const dispatch = createEventDispatcher();
  
  const sizeClasses = {
    sm: 'max-w-md',
    md: 'max-w-lg',
    lg: 'max-w-2xl',
    xl: 'max-w-4xl'
  };
  
  function handleClose() {
    dispatch('close');
  }
  
  function handleKeydown(event) {
    if (event.key === 'Escape') {
      handleClose();
    }
  }
</script>

<svelte:window on:keydown={handleKeydown} />

<div 
  class="fixed inset-0 bg-gray-600 bg-opacity-50 overflow-y-auto h-full w-full z-50 flex items-center justify-center p-4"
  transition:fade
  on:click={handleClose}
>
  <div 
    class="relative bg-white rounded-lg shadow-xl {sizeClasses[size]} w-full"
    transition:scale
    on:click|stopPropagation
  >
    <!-- 头部 -->
    {#if title}
      <div class="px-6 py-4 border-b border-gray-200">
        <div class="flex items-center justify-between">
          <h3 class="text-lg font-medium text-gray-900">{title}</h3>
          <button 
            on:click={handleClose}
            class="text-gray-400 hover:text-gray-600"
          >
            <svg class="w-6 h-6" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path>
            </svg>
          </button>
        </div>
      </div>
    {/if}
    
    <!-- 内容 -->
    <div class="px-6 py-4">
      <slot />
    </div>
  </div>
</div>
```
