// NAT穿越尝试
func (c *Conn) createEndpoints() ([]tailcfg.Endpoint, error) {
    var eps []tailcfg.Endpoint
    
    // 1. 本地接口地址
    addrs, _ := interfaces.LocalAddresses()
    for _, addr := range addrs {
        eps = append(eps, tailcfg.Endpoint{
            Addr: netip.AddrPortFrom(addr, c.port),
            Type: tailcfg.EndpointLocal,
        })
    }
    
    // 2. STUN发现的公网地址
    if stunAddr, err := c.stunClient.Discover(); err == nil {
        eps = append(eps, tailcfg.Endpoint{
            Addr: stunAddr,
            Type: tailcfg.EndpointSTUN,
        })
    }
    
    // 3. UPnP映射
    if upnpAddr, err := c.upnpClient.AddPortMapping(); err == nil {
        eps = append(eps, tailcfg.Endpoint{
            Addr: upnpAddr,
            Type: tailcfg.EndpointUPnP,
        })
    }
    
    return eps, nil
}

```go
// DERP消息转发
func (s *Server) forwardPacket(src, dst key.NodePublic, packet []byte) error {
    s.clientsMu.RLock()
    dstClient, ok := s.clients[dst]
    s.clientsMu.RUnlock()
    
    if !ok {
        return errors.New("destination not connected")
    }
    
    // 构造转发消息
    msg := &derp.RecvPacket{
        From:   src,
        Data:   packet,
    }
    
    // 发送给目标客户端
    return dstClient.send(msg)
}

func (c *Client) keepAlive() {
    ticker := time.NewTicker(60 * time.Second)
    defer ticker.Stop()
    
    for {
        select {
        case <-ticker.C:
            if err := c.sendPing(); err != nil {
                c.logf("DERP ping failed: %v", err)
                c.reconnect()
            }
        case <-c.done:
            return
        }
    }

// 本地网络扫描
func (m *Monitor) pollInterfaceState() (*interfaces.State, error) {
    st := &interfaces.State{}
    
    // 获取网络接口
    ifaces, err := net.Interfaces()
    if err != nil {
        return nil, err
    }
    
    for _, iface := range ifaces {
        if iface.Flags&net.FlagUp == 0 {
            continue
        }
        
        addrs, err := iface.Addrs()
        if err != nil {
            continue
        }
        
        for _, addr := range addrs {
            if ipnet, ok := addr.(*net.IPNet); ok {
                st.InterfaceIPs = append(st.InterfaceIPs, ipnet.IP)
            }
        }
    }
    
    return st, nil
}

type Resolver struct {
    logf        logger.Logf
    netMap      *netmap.NetworkMap
    
    // DNS配置
    config      *dns.Config
    
    // 上游DNS
    upstream    []netip.AddrPort

}

// DNS查询处理
func (r *Resolver) resolve(name string, qtype dns.Type) (*dns.Msg, error) {
    // 1. 检查本地hosts
    if addr, ok := r.hosts[name]; ok {
        return r.makeResponse(name, qtype, addr), nil
    }
    
    // 2. 检查Tailscale域名
    if strings.HasSuffix(name, ".ts.net") {
        return r.resolveTailscale(name, qtype)
    }
    
    // 3. 转发到上游DNS
    return r.forwardUpstream(name, qtype)
}
```

```go
// ACL规则评估
func (p *Policy) evalACL(src, dst netip.AddrPort, proto ipproto.Proto) bool {
    srcNode := p.findNode(src.Addr())
    if srcNode == nil {
        return false
    }
    
    for _, rule := range p.ACLs {
        if p.matchSource(rule.Src, srcNode) &&
           p.matchDestination(rule.Dst, dst, proto) {
            return rule.Action == "accept"
        }
    }
    
    return false // 默认拒绝
}

// 标签匹配
func (p *Policy) matchTags(ruleTags, nodeTags []string) bool {
    for _, ruleTag := range ruleTags {
        for _, nodeTag := range nodeTags {
            if ruleTag == nodeTag || ruleTag == "*" {
                return true
            }
        }
    }
    return false
}

// 零拷贝数据包处理
func (c *Conn) sendPacket(packet []byte) error {
    // 使用内存池避免分配
    buf := c.bufPool.Get().([]byte)
    defer c.bufPool.Put(buf)
    
    // 直接写入网络缓冲区
    n := copy(buf, packet)
    return c.rawConn.Write(buf[:n])
}

// 批量数据包处理
func (c *Conn) sendBatch(packets [][]byte) error {
    msgs := make([]syscall.RawSockaddrInet4, len(packets))
    
    for i, packet := range packets {
        msgs[i] = syscall.RawSockaddrInet4{
            Family: syscall.AF_INET,
            Data:   packet,
        }
    }
    
    return syscall.Sendmmsg(c.fd, msgs, 0)
}
```
